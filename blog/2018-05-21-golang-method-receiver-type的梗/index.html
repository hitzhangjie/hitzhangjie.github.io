golang method receiver-typeзҡ„жў—

иҝҷйҮҢжқҘиҒҠиҒҠmethod receiver typeдёәд»Җд№ҲдёҚиғҪжҳҜpointerе’Ңinterfaceзұ»еһӢгҖӮ

1 receiver-typeеҝ…йЎ»ж»Ўи¶ізҡ„жқЎд»¶

golangйҮҢйқўжҸҗдҫӣдәҶдёҖе®ҡзҡ„йқўеҗ‘еҜ№иұЎж”ҜжҢҒпјҢжҜ”еҰӮжҲ‘д»¬еҸҜд»Ҙдёәзұ»еһӢTжҲ–иҖ…*Tе®ҡд№үжҲҗе‘ҳж–№жі•пјҲзұ»еһӢTз§°дёәжҲҗе‘ҳж–№жі•зҡ„receiver-typeпјүпјҢдҪҶжҳҜиҝҷйҮҢзҡ„зұ»еһӢTеҝ…йЎ»ж»Ўи¶іеҰӮдёӢеҮ дёӘжқЎд»¶пјҡ

  • Tеҝ…йЎ»жҳҜе·Із»Ҹе®ҡд№үиҝҮзҡ„зұ»еһӢпјӣ
  • TдёҺеҪ“еүҚж–№жі•е®ҡд№үеҝ…йЎ»еңЁеҗҢдёҖдёӘpackageдёӢйқўпјӣ
  • TдёҚиғҪжҳҜжҢҮй’Ҳпјӣ
  • TдёҚиғҪжҳҜжҺҘеҸЈзұ»еһӢпјӣ

еүҚйқўдёӨзӮ№йғҪжҜ”иҫғе®№жҳ“зҗҶи§ЈпјҢдёӢйқўдёӨзӮ№жҳҜд»Җд№Ҳжў—пјҹдёәд»Җд№Ҳе°ұдёҚиғҪеңЁжҢҮй’Ҳзұ»еһӢдёҠж·»еҠ ж–№жі•пјҹдёәд»Җд№Ҳе°ұдёҚиғҪеңЁinterfaceдёҠж·»еҠ ж–№жі•пјҹеҪ“з„¶еҸҜд»ҘдёҖеҸҘиҜқеҫ…иҝҮпјҢgolangдёҚж”ҜжҢҒпјҢдҪҶжҳҜжҲ‘жғій—®дёӢдёәд»Җд№Ҳпјҹ

2 receiver-typeдёәд»Җд№ҲдёҚиғҪжҳҜжҢҮй’Ҳзұ»еһӢпјҹ

golangе…Ғи®ёдёә зұ»еһӢжҢҮй’Ҳ*T ж·»еҠ ж–№жі•пјҢдҪҶжҳҜдёҚе…Ғи®ёдёә жҢҮй’Ҳзұ»еһӢжң¬иә« ж·»еҠ ж–№жі•гҖӮжҢүзҺ°жңүgolangзҡ„е®һзҺ°ж–№ејҸпјҢдёәжҢҮй’Ҳзұ»еһӢж·»еҠ ж–№жі•дјҡеҜјиҮҙж–№жі•и°ғз”Ёж—¶зҡ„жӯ§д№үгҖӮ

зңӢдёӢйқўиҝҷдёӘзӨәдҫӢзЁӢеәҸгҖӮ

type T int 
func (t *T) Get() T { 
    return *t + 1 
} 
type P *T 
func (p P) Get() T { 
    return *p + 2 
} 
func F() { 
    var v1 T 
    var v2 = &v1 
    var v3 P = &v1 
    fmt.Println(v1.Get(), v2.Get(), v3.Get()) 
}

зӨәдҫӢзЁӢеәҸдёӯ v3.Get() еӯҳеңЁи°ғз”Ёжӯ§д№үпјҢзј–иҜ‘еҷЁдёҚзҹҘйҒ“иҜҘи°ғз”Ёе“ӘдёӘж–№жі•дәҶгҖӮеҰӮжһңиҰҒж”ҜжҢҒеңЁжҢҮй’Ҳиҝҷз§Қreceiver-typeдёҠе®ҡд№үж–№жі•пјҢgolangзј–иҜ‘еҷЁеҠҝеҝ…иҰҒе®һзҺ°ең°жӣҙеӨҚжқӮжүҚиғҪж”ҜжҢҒеҲ°пјҢжҢҮй’Ҳжң¬жқҘе°ұжҜ”иҫғе®№жҳ“з ҙеқҸеҸҜиҜ»жҖ§пјҢиҝҳиҰҒеңЁдёҖз§ҚжҢҮй’Ҳзұ»еһӢдёҠе®ҡд№үж–№жі•пјҢеҜ№дҪҝз”ЁиҖ…гҖҒзј–иҜ‘еҷЁејҖеҸ‘иҖ…иҖҢиЁҖеҸҜиғҪйғҪжҳҜд»¶иҙ№еҠӣдёҚи®ЁеҘҪзҡ„дәӢжғ…гҖӮ

3 receiver-typeдёәд»Җд№ҲдёҚиғҪжҳҜжҺҘеҸЈзұ»еһӢпјҹ

иҝҷжІЎжңүд»Җд№ҲеҘҪжҸЈжөӢзҡ„пјҢеҸӘжҳҜgolang runtimeдёҚж”ҜжҢҒиҖҢе·ІгҖӮgolangзҺ°еңЁзҡ„е®һзҺ°йҮҢпјҢinterfaceеҶ…йғЁз»“жһ„еҸӘиғҪиЎЁзӨәж–№жі•еҺҹеһӢпјҢдҪҶжҳҜдёҚеҢ…жӢ¬ж–№жі•е®ҡд№үпјҢstructжүҚеҸҜд»ҘеҢ…жӢ¬ж–№жі•е®ҡд№үпјҲиҝҷйғЁеҲҶеҶ…е®№ж„ҹе…ҙи¶Јзҡ„еҸҜд»Ҙзҝ»дёӢgolangзҡ„жәҗз ҒпјҢиҝҷйҮҢдёҚеұ•ејҖдәҶпјүгҖӮ

еҪ“дёҖдёӘзұ»еһӢе®һзҺ°дәҶжҹҗдёӘжҺҘеҸЈеЈ°жҳҺзҡ„е…ЁйғЁж–№жі•ж—¶пјҢе°ұиҜҙиҝҷдёӘзұ»еһӢе®һзҺ°дәҶиҝҷдёӘжҺҘеҸЈпјҢе°ұеҸҜд»Ҙе°ҶиҝҷдёӘзұ»еһӢзҡ„еҖјиөӢеҖјз»ҷиҜҘжҺҘеҸЈзұ»еһӢзҡ„еҖјпјҢжӯӨж—¶дјҡжӣҙж–°жҺҘеҸЈеҖјзҡ„ dynamic valueе’Ңdynamic type еӯ—ж®өгҖӮиҝҷйҮҢйңҖиҰҒ ж №жҚ®жҺҘеҸЈдёӯжҳҜеҗҰе®ҡд№үдәҶж–№жі•еҲ—иЎЁ жқҘиҝӣдёҖжӯҘеҲҶжһҗпјҢеҸҜд»Ҙз»ҶеҲҶдёә жҺҘеҸЈж–№жі•еҲ—иЎЁдёәз©ә е’Ң жҺҘеҸЈж–№жі•еҲ—иЎЁдёҚз©ә дёӨз§Қжғ…еҶөи®Ёи®әгҖӮ

дёӢйқўд»ҘиҝҷйҮҢзҡ„зӨәдҫӢд»Јз ҒдёәдҫӢиҝӣиЎҢеҲҶжһҗпјҡ

// - жҺҘеҸЈзұ»еһӢ1
type EmptyIface interface{
}
// - жҺҘеҸЈзұ»еһӢ2
type Stringer interface {
    String() string
}

// - зұ»еһӢBinaryе®һзҺ°дәҶжҺҘеҸЈStringerпјҢд№ҹе®һзҺ°дәҶEmptyIface
type Binary uint64

func (i Binary) String() string {
    return strconv.Uitob64(i.Get(), 2)
}

func (i Binary) Get() uint64 {
    return uint64(i)
}

// - дёәжҺҘеҸЈеҖјиөӢеҖј
var b Binary = Binary(1)
var eface EmptyIface = b
var iface Stringer = b
fmt.Println(iface)

3.1 з©әжҺҘеҸЈinterface#

efaceеҜ№еә”жҺҘеҸЈзұ»еһӢжҳҜinterface{}пјҢе№¶дё”BinaryеҖјbе°ҸдәҺзӯүдәҺsizeof(uintptr)пјҢйӮЈд№Ҳefaceзҡ„dynamic valueе°ұжҳҜзӣҙжҺҘжӢ·иҙқbпјӣеҰӮжһңиҝҷйҮҢжҳҜeface := anyBigStruct {….}пјҢе№¶дё”anyBigStruct sizeеӨ§дәҺsizeof(uintptr)пјҢйӮЈд№Ҳе°ұйңҖиҰҒејҖиҫҹеҶ…еӯҳжӢ·иҙқanyBigStructзҡ„еҖјпјҢе№¶е°Ҷж–°ејҖиҫҹеҶ…еӯҳзҡ„ең°еқҖи®ҫзҪ®еҲ°dynamic valueеӯ—ж®өгҖӮdynamic typeе°ұжҢҮеҗ‘еҜ№еә”зұ»еһӢзҡ„е®ҡд№үдәҶгҖӮ

жӯӨж—¶зҡ„жҺҘеҸЈзұ»еһӢеҸҜд»ҘиЎЁзӨәдёәпјҡ

type eface struct {
	_type *_type
	data  unsafe.Pointer
}

жӯӨж—¶зҡ„жҺҘеҸЈеҖјеҸҜд»ҘиЎЁзӨәдёәпјҡ

3.2 йқһз©әжҺҘеҸЈinterface{ method-list }#

еҰӮжһңifaceеҜ№еә”жҺҘеҸЈзұ»еһӢinterface{methods-list}пјҢиҝҷдёӘж—¶еҖҷdynamic valueиҝҳжҳҜдёҺ1пјүдёӯеҗҢж ·зҡ„еӨ„зҗҶпјҢдҪҶжҳҜdynamic typeеӨ„зҗҶж–№ејҸдёҚеҗҢпјҢйңҖиҰҒеҜ№иҝҗиЎҢж—¶жҺҘеҸЈж–№жі•зҡ„и°ғз”Ёең°еқҖиҝӣиЎҢеӨ„зҗҶгҖӮ

иҝҷж—¶дјҡеҲӣе»әдёҖдёӘ itableпјҲзұ»дјјc++иҷҡеҮҪж•°vtableпјүпјҢиҝҷдёӘitableйҮҢйқўдҝқеӯҳдәҶifaceзҡ„жҺҘеҸЈзұ»еһӢд»ҘеҸҠжҺҘеҸЈдёӯеЈ°жҳҺзҡ„еҗ„дёӘж–№жі•еҺҹеһӢеҜ№еә”зҡ„и°ғз”Ёең°еқҖпјҢи°ғз”Ёең°еқҖд»ҺдҪ•иҖҢжқҘпјҹиҝҷйҮҢзҡ„и°ғз”Ёең°еқҖд№ҹе°ұжҳҜжҺҘеҸЈеҖјеҠЁжҖҒзұ»еһӢдёӯе®һзҺ°зҡ„ж–№жі•зҡ„и°ғз”Ёең°еқҖгҖӮи®ҫзҪ®е®ҢжҺҘеҸЈж–№жі•зҡ„жүҖжңүи°ғз”Ёең°еқҖеҗҺпјҢitableжһ„е»әе®ҢжҲҗпјҢз„¶еҗҺе°ҶifaceжҺҘеҸЈеҖјйҮҢйқўзҡ„tabеӯ—ж®өжҢҮеҗ‘иҜҘitableгҖӮ

жӯӨж—¶зҡ„жҺҘеҸЈзұ»еһӢеҸҜд»ҘиЎЁзӨәдёәпјҡ

type iface struct {
	tab  *itab
	data unsafe.Pointer
}

жӯӨж—¶зҡ„жҺҘеҸЈеҖјеҸҜд»ҘиЎЁзӨәдёәпјҡ

3.3 жҺҘеҸЈж–№жі•и°ғз”Ё#

йҖҡиҝҮжҺҘеҸЈеҖји°ғз”Ёж–№жі•зҡ„ж—¶еҖҷпјҢйңҖиҰҒж №жҚ® вҖңиҜҘжҺҘеҸЈеҖјзұ»еһӢгҖҒжҺҘеҸЈеҖјеҜ№еә”зҡ„еҠЁжҖҒзұ»еһӢвҖқ жҹҘиҜўеҜ№еә”зҡ„жҺҘеҸЈж–№жі•е®һйҷ…зҡ„и°ғз”Ёең°еқҖгҖӮеҰӮжһңиҜҘжҺҘеҸЈзұ»еһӢйҮҢйқўж №жң¬е°ұжІЎжңүж–№жі•еҲ—иЎЁпјҢйӮЈиӮҜе®ҡе°ұжҠҘй”ҷдәҶд№ҹдёҚдјҡжҹҘиҜўtabпјҢefaceйҮҢйқўд№ҹжІЎиҝҷдёӘеӯ—ж®өпјӣеҰӮжһңжҺҘеҸЈзұ»еһӢйҮҢйқўжңүиҝҷдёӘж–№жі•е®ҡд№үпјҢйӮЈе°ұж №жҚ®iface.tabжҢҮеҗ‘зҡ„itableеҺ»жҹҘиҜўеҜ№еә”иҜҘжҺҘеҸЈзұ»еһӢгҖҒеҠЁжҖҒзұ»еһӢзҡ„еҜ№еә”ж–№жі•зҡ„и°ғз”Ёең°еқҖпјҢз„¶еҗҺжү§иЎҢзӣ®ж Үең°еқҖеӨ„зҡ„ж–№жі•дҪ“гҖӮ

д»Һе°ҶBinaryеҖјbиөӢеҖјз»ҷifaceпјҢеҶҚеҲ°йҖҡиҝҮifaceжҹҘиҜўеҠЁжҖҒзұ»еһӢBinaryдёӯе®һзҺ°зҡ„ж–№жі•Stringerи°ғз”Ёең°еқҖпјҢжңҖз»Ҳжү§иЎҢBinaryдёӯе®һзҺ°зҡ„ж–№жі•String()пјҢиҝҷж•ҙдёӘйҖ»иҫ‘еӨ„зҗҶиҝҮзЁӢдёӯеҰӮдҪ•еҜ№жҺҘеҸЈзұ»еһӢеЈ°жҳҺж–№жі•еҲ—иЎЁгҖҒеҠЁжҖҒзұ»еһӢе®һзҺ°ж–№жі•еҲ—иЎЁиҝӣиЎҢеӨ„зҗҶпјҢжҲ‘д»¬еә”иҜҘжё…жҘҡдәҶгҖӮ

жҖ»иҖҢиЁҖд№ӢпјҢзҺ°еңЁgolangе®һзҺ°дёӯinterfaceзҡ„еҶ…йғЁиЎЁзӨәдёҚеҢ…еҗ«вҖңж–№жі•е®ҡд№үвҖқзӣёе…ізҡ„еӯ—ж®өпјҢж— жі•иЎЁзӨәдёәinterfaceж·»еҠ зҡ„ж–№жі•е®ҡд№үпјҢзј–иҜ‘еҷЁеҪ“з„¶д№ҹдёҚе…Ғи®ёгҖӮ

дёҚеҗҢзҡ„зј–зЁӢиҜӯиЁҖеҜ№еҮҪж•°и°ғз”Ёзҡ„еӨҡжҖҒе®һзҺ°жңүдёҚеҗҢзҡ„жҖқи·ҜпјҢc++жҳҜйҖҡиҝҮеЎ«е……еҹәжң¬зұ»еһӢзҡ„иҷҡеҮҪж•°иЎЁзҡ„ж–№ејҸпјҢgolangжҳҜйҖҡиҝҮзұ»дјјзҡ„иҝҷж ·дёҖз§ҚжҹҘиҜўиЎЁзҡ„еҪўејҸгҖӮжҖ»з»“дёҖдёӢе°ұжҳҜе№¶йқһgolangж— жі•йҖҡиҝҮжү©еұ•ж”ҜжҢҒеңЁжҺҘеҸЈдёҠж·»еҠ ж–№жі•пјҢеҸӘжҳҜиҝҷж ·жҳҜеҗҰзңҹзҡ„жңүеҝ…иҰҒпјҢеҖјеҫ—е•ҶжҰ·гҖӮиҮіе°‘йқһз©әжҺҘеҸЈжҳҜз”ЁжқҘе®ҡд№үдёҖз§ҚжҳҺзЎ®зҡ„иЎҢдёәпјҲcontractпјүзҡ„пјҢеңЁдёҠйқўеҸҲеҠ дёӘж–№жі•пјҢжҳҜд»Җд№Ҳж„ҸжҖқе‘ўпјҹеӣ жӯӨдёҚе…Ғи®ёе°ҶжҺҘеҸЈзұ»еһӢе®ҡд№үдёәreceiver typeд№ҹжҳҜдёҺжҺҘеҸЈзҡ„е®ҡдҪҚзӣёеҜ№еә”зҡ„еҗ§гҖӮ

4 жҖ»з»“

receiver-typeдёҚе…Ғи®ёдёәжҢҮй’Ҳзұ»еһӢе’ҢжҺҘеҸЈзұ»еһӢпјҢеҜ№е…¶дёӯеҺҹеӣ иҝӣиЎҢдәҶдёҖзӮ№жҖқиҖғе’ҢжҖ»з»“гҖӮ
